Question 1
A number increased by 20% becomes 120.
Find the original number.
Solution:
	Original number=x
100% is actual
20% of x=20/100x
	=1/5x
Original number+20% of original number=120
X+1/5x=120
5x+1x/5=120
6x/5=120*5
6x=600
X=600/6
X=100

Find the missing number in the array
[1,2,3,4,6,7,8]
MISSING ELEMENT
Solution(JAVA)

	import java.util.Scanner
	public class Main{
	public static void main(String[]){
	Scanner sc = new Scanner(System.in);
	int n=sc.nextInt();
	int arr[]=new int[n-1];
	int sum=0;
	for(int i=0;i<arr.length;i++){
		arr[i]=sc.nextInt();
		sum=sum+arr[i];
	}
	int total=n*(n+1)/2;
	int miss=total-sum;
	System.out.println(miss);
}
}

      =8*9/2

	=36-31

Find the maximum element in the array
[3,8,1,10,4]

import java.util.Scanner;
public class Main {
public static void main(String[] args) {
Scanner sc = new Scanner(System.in);
int n = sc.nextInt();
int arr[] = new int[n];
for(int i = 0; i < n; i++)
{
	arr[i] = sc.nextInt();
}
int left = 0;
int right = n - 1;
int max = arr[0];
while(left <= right){
	if(arr[left] > max){
		max = arr[left];
}
	if(arr[right] > max){
	max = arr[right];
}
	left++;
	right--;
}
System.out.println("Maximum element: " + max);
}
}
